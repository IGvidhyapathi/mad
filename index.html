1. Profit of all months (Line Plot)

import pandas as pd import matplotlib.pyplot as plt df = pd.read_csv("CSV file path") months = df['month_number'].tolist() profits = df['total_profit'].tolist() plt.plot(months, profits, marker='o', color='b', linestyle='-', linewidth=2, markersize=6) plt.title('Total Profit of All Months') plt.xlabel('Months') plt.ylabel('Profit ($)') plt.grid(True) plt.show()

2. Toothpaste sales data (Scatter Plot)

import pandas as pd import matplotlib.pyplot as plt df = pd.read_csv("CSV path here") months = df['month_number'].tolist() sales = df['toothpaste'].tolist() plt.scatter(months, sales, color='r', marker='o') plt.title('Toothpaste Sales per Month') plt.xlabel('Months') plt.ylabel('Sales (Units)') plt.grid(True) plt.show()

3. Facewash and Facecream (Bar Chart)

import pandas as pd import matplotlib.pyplot as plt df = pd.read_csv("CSV here") plt.bar(['Face Cream', 'Face Wash'], [df['facecream'].sum(), df['facewash'].sum()], color=['blue', 'green']) plt.title('Total Sales') plt.xlabel('Products') plt.ylabel('Units Sold') plt.show()

4. Profit of each month (Histogram)

import pandas as pd import matplotlib.pyplot as plt df = pd.read_csv("CSV here") plt.hist(df['total_profit'], bins=5, color='skyblue', edgecolor='black') plt.title('Distribution of Monthly Profits') plt.xlabel('Profit Range ($)') plt.ylabel('Frequency') plt.show()

5. Sales distribution (Pie Chart)

import pandas as pd import matplotlib.pyplot as plt df = pd.read_csv("CSV here") # Extract product names and their total sales products = ['Face Cream', 'Face Wash', 'Toothpaste', 'Bathing Soap'] sales_data = [df['facecream'].sum(), df['facewash'].sum(), df['toothpaste'].sum(), df['bathingsoap'].sum()] # Create the pie chart plt.pie(sales_data, labels=products, autopct='%1.1f%%', startangle=140, colors=['lightblue', 'lightgreen', 'lightcoral', 'lightyellow']) plt.title('Total Sales for Each Product (Last Year)') plt.show()
